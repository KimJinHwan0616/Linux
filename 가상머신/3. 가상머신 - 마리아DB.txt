-------------------MariaDB 환경 설정-------------------

create database bigdata;	# 데이터베이스(bigdata) 생성

##
계정 추가

create user 사용자@접속위치
identified by 비밀번호
##

create user 'bigdata'@'%'
identified by '1234';

##
권한 부여

grant 권한 on
데이터베이스.테이블 to 사용자@접속위치
##

grant all privileges on
bigdata.* to 'bigdata'@'%';

flush privileges;
exit
